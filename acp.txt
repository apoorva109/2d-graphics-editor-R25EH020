#include <stdio.h>
#include <stdlib.h>

#define ROWS 20
#define COLS 60
#define MAX 10

// Canvas stored as 2D character array
char canvas[ROWS][COLS];

// Shape storage
int shape_type[MAX];   // 1=line, 2=rect, 3=circle, 4=triangle
int shape_data[MAX][6]; // stores coordinates
int shape_count = 0;

// Fill canvas with underscores
void init_canvas() {
    int i, j;
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            canvas[i][j] = '_';
        }
    }
}

// Display the canvas
void display_canvas() {
    int i, j;
    printf("\n");
    for (i = 0; i < ROWS; i++) {
        for (j = 0; j < COLS; j++) {
            printf("%c", canvas[i][j]);
        }
        printf("\n");
    }
}

// Plot a point safely
void plot(int x, int y) {
    if (x >= 0 && x < COLS && y >= 0 && y < ROWS) {
        canvas[y][x] = '*';
    }
}

// Draw a line using Bresenham's algorithm
void draw_line(int x1, int y1, int x2, int y2) {
    int dx = abs(x2 - x1);
    int dy = abs(y2 - y1);
    int sx = (x1 < x2) ? 1 : -1;
    int sy = (y1 < y2) ? 1 : -1;
    int err = dx - dy;
    int e2;

    while (1) {
        plot(x1, y1);
        if (x1 == x2 && y1 == y2) break;
        e2 = 2 * err;
        if (e2 > -dy) { err -= dy; x1 += sx; }
        if (e2 < dx)  { err += dx; y1 += sy; }
    }
}

// Draw a rectangle outline
void draw_rectangle(int x, int y, int w, int h) {
    int i;
    for (i = 0; i < w; i++) {
        plot(x + i, y);
        plot(x + i, y + h - 1);
    }
    for (i = 0; i < h; i++) {
        plot(x, y + i);
        plot(x + w - 1, y + i);
    }
}

// Draw a circle using midpoint circle algorithm
void draw_circle(int cx, int cy, int r) {
    int x = 0, y = r;
    int d = 3 - 2 * r;

    while (y >= x) {
        plot(cx + x, cy + y); plot(cx - x, cy + y);
        plot(cx + x, cy - y); plot(cx - x, cy - y);
        plot(cx + y, cy + x); plot(cx - y, cy + x);
        plot(cx + y, cy - x); plot(cx - y, cy - x);
        x++;
        if (d > 0) { y--; d = d + 4 * (x - y) + 10; }
        else       { d = d + 4 * x + 6; }
    }
}

// Draw a triangle using 3 lines
void draw_triangle(int x1, int y1, int x2, int y2, int x3, int y3) {
    draw_line(x1, y1, x2, y2);
    draw_line(x2, y2, x3, y3);
    draw_line(x3, y3, x1, y1);
}

// Redraw all shapes
void redraw() {
    int i;
    init_canvas();
    for (i = 0; i < shape_count; i++) {
        int *d = shape_data[i];
        if (shape_type[i] == 1) draw_line(d[0], d[1], d[2], d[3]);
        if (shape_type[i] == 2) draw_rectangle(d[0], d[1], d[2], d[3]);
        if (shape_type[i] == 3) draw_circle(d[0], d[1], d[2]);
        if (shape_type[i] == 4) draw_triangle(d[0], d[1], d[2], d[3], d[4], d[5]);
    }
}

// Add a shape
void add_shape() {
    if (shape_count >= MAX) {
        printf("Canvas is full!\n");
        return;
    }

    int choice;
    printf("\n1. Line\n2. Rectangle\n3. Circle\n4. Triangle\n");
    printf("Choose shape: ");
    scanf("%d", &choice);

    int *d = shape_data[shape_count];
    shape_type[shape_count] = choice;

    if (choice == 1) {
        printf("Enter x1 y1 x2 y2: ");
        scanf("%d %d %d %d", &d[0], &d[1], &d[2], &d[3]);
    } else if (choice == 2) {
        printf("Enter x y width height: ");
        scanf("%d %d %d %d", &d[0], &d[1], &d[2], &d[3]);
    } else if (choice == 3) {
        printf("Enter center_x center_y radius: ");
        scanf("%d %d %d", &d[0], &d[1], &d[2]);
    } else if (choice == 4) {
        printf("Enter x1 y1 x2 y2 x3 y3: ");
        scanf("%d %d %d %d %d %d", &d[0], &d[1], &d[2], &d[3], &d[4], &d[5]);
    } else {
        printf("Invalid choice!\n");
        return;
    }

    shape_count++;
    redraw();
    printf("Shape added! ID = %d\n", shape_count);
}

// List all shapes
void list_shapes() {
    int i;
    if (shape_count == 0) {
        printf("No shapes added yet.\n");
        return;
    }
    printf("\n--- Shape List ---\n");
    for (i = 0; i < shape_count; i++) {
        int *d = shape_data[i];
        printf("ID %d: ", i + 1);
        if (shape_type[i] == 1) printf("Line (%d,%d) to (%d,%d)\n", d[0], d[1], d[2], d[3]);
        if (shape_type[i] == 2) printf("Rectangle at (%d,%d) w=%d h=%d\n", d[0], d[1], d[2], d[3]);
        if (shape_type[i] == 3) printf("Circle center=(%d,%d) r=%d\n", d[0], d[1], d[2]);
        if (shape_type[i] == 4) printf("Triangle (%d,%d),(%d,%d),(%d,%d)\n", d[0], d[1], d[2], d[3], d[4], d[5]);
    }
    printf("------------------\n");
}

// Delete a shape by ID
void delete_shape() {
    int id, i;
    list_shapes();
    if (shape_count == 0) return;

    printf("Enter ID to delete: ");
    scanf("%d", &id);
    id--; // convert to index

    if (id < 0 || id >= shape_count) {
        printf("Invalid ID!\n");
        return;
    }

    for (i = id; i < shape_count - 1; i++) {
        shape_type[i] = shape_type[i + 1];
        int j;
        for (j = 0; j < 6; j++) shape_data[i][j] = shape_data[i + 1][j];
    }
    shape_count--;
    redraw();
    printf("Shape deleted!\n");
}

// Modify a shape by ID
void modify_shape() {
    int id;
    list_shapes();
    if (shape_count == 0) return;

    printf("Enter ID to modify: ");
    scanf("%d", &id);
    id--; // convert to index

    if (id < 0 || id >= shape_count) {
        printf("Invalid ID!\n");
        return;
    }

    int *d = shape_data[id];
    printf("Enter new values:\n");

    if (shape_type[id] == 1) {
        printf("Enter x1 y1 x2 y2: ");
        scanf("%d %d %d %d", &d[0], &d[1], &d[2], &d[3]);
    } else if (shape_type[id] == 2) {
        printf("Enter x y width height: ");
        scanf("%d %d %d %d", &d[0], &d[1], &d[2], &d[3]);
    } else if (shape_type[id] == 3) {
        printf("Enter center_x center_y radius: ");
        scanf("%d %d %d", &d[0], &d[1], &d[2]);
    } else if (shape_type[id] == 4) {
        printf("Enter x1 y1 x2 y2 x3 y3: ");
        scanf("%d %d %d %d %d %d", &d[0], &d[1], &d[2], &d[3], &d[4], &d[5]);
    }

    redraw();
    printf("Shape modified!\n");
}

int main() {
    int choice;
    init_canvas();

    while (1) {
        printf("\n=== 2D Graphics Editor ===\n");
        printf("1. Add Shape\n");
        printf("2. Delete Shape\n");
        printf("3. Modify Shape\n");
        printf("4. Display Canvas\n");
        printf("5. List Shapes\n");
        printf("6. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        if (choice == 1) add_shape();
        else if (choice == 2) delete_shape();
        else if (choice == 3) modify_shape();
        else if (choice == 4) display_canvas();
        else if (choice == 5) list_shapes();
        else if (choice == 6) { printf("Goodbye!\n"); break; }
        else printf("Invalid choice!\n");
    }

    return 0;
}
